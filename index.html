// initialize your couchApp
    couchApp = require('couchapp');
    couchDB_endpoint = 'http://corscouch.example.com';
    app = new couchApp(couchDB_endpoint);
    
    // 
    // Account 
    //
    
    // sign up
    app.account.sign_up('joe@example.com', 'secret')
      .done( function(user) { } ) 
      .fail( function(err)  { } )
      
    // sign in
    app.account.sign_in('joe@example.com', 'secret')
      .done( function(user) { } ) 
      .fail( function(err)  { } )
      
    // sign out
    app.account.sign_out()
      .done( function(user) { } ) 
      .fail( function(err)  { } )
    
    // change password (coming soon)
    app.account.change_password('current_secret', 'new_secret')
      .done( function(user) { } ) 
      .fail( function(err)  { } )
      
    // forgot password (coming soon)
    app.account.forgot_password('joe@example.com')
      .done( function(user) { } ) 
      .fail( function(err)  { } )
      
    //
    // Store
    //
    
    // helper to generate uuids
    uuid = app.store.uuid(length)
    
    // create a new object
    type = 'couch'
    attributes = {color: "red"}
    app.store.save(type , attributes )
      .done ( function(new_object) { } )
      .fail ( function(err)        { } )
      
    // update an existing object
    id   = 'abc4567'
    type = 'rule'
    attributes  = {color: "red", name: "relax"}
    app.store.save( type, id, attributes )
      .done ( function(new_object) { } )
      .fail ( function(err)        { } )
      
    // alternative syntax
    app.store.save( {name: "relax", type: "couch", id: "abc4567"} )
    
    // load one object
    type = 'couch'
    id   = 'abc4567'
    app.store.load( type, id )
      .done ( function(object) { } )
      .fail ( function(err)    { } )
      
    // Load all objects
    app.store.loadAll()
      .done ( function(objects) { } )
      .fail ( function(err)     { } )
      
    // Load all objects from one type
    type = 'couch'
    app.store.loadAll( type )
      .done ( function(objects) { } )
      .fail ( function(err)     { } )
      
    // delete an existing object
    type = 'couch'
    id   = 'abc4567'
    app.store.delete( type, id )
      .done ( function(deleted_object) { } )
      .fail ( function(err)            { } )
      
    //
    // Synchronization (coming soon)
    //
    // When signed in, local changes do get synched automatically.
    // You can subscribe to remote updates
    // 
    
    // new doc created
    app.remote.on_create( function( created_object) { } )

    // existing doc updated
    app.remote.on_update( function( update_object)  { } )

    // doc deleted
    app.remote.on_delete( function( deleted_object) { } )

    // any of above events
    app.remote.on_change( function( changed_object) { } )

    // all listeners can be filtered by type
    app.remote.on_create( type, function( created_object) { } )
    app.remote.on_update( type, function( update_object)  { } )
    app.remote.on_delete( type, function( deleted_object) { } )
    app.remote.on_change( type, function( changed_object) { } )
