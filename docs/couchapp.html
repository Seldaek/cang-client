    sign_up : (email, password) ->
      prefix  = 'org.couchdb.user'
      key     = "#{prefix}:#{email}"
    
      user = 
        _id           : key
        name          : email
        type          : 'user'
        roles         : []
        password      : password
    
      $.ajax
        type        : 'PUT'
        url         : "#{@couchDB_url}/_users/#{encodeURIComponent key}"
        data        : JSON.stringify user
        contentType : "application/json"
